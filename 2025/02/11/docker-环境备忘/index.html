docker环境备忘

安装docker

curl -sSL https://get.daocloud.io/docker | sh

service docker start
  • [官方某版]安装docker-compose *
curl -L https://get.daocloud.io/docker/compose/releases/download/1.25.5/docker-compose-`uname -s`-`uname -m` > /usr/bin/docker-compose

chmod +x /usr/bin/docker-compose

[实测带依赖]安装docker/docker-compose

sudo apt-get install python3.7-distutils   3.7 是对应你的版本
curl -s https://bootstrap.pypa.io/get-pip.py | python3
curl -s https://get.docker.com/ | sh
systemctl start docker
pip install docker-compose

安装ping

apt-get install inetutils-ping net-tools

docker run -itd --name hfish \
-v /usr/share/hfish:/usr/share/hfish \
--network host \
--privileged=true \
threatbook/hfish-server:latest

awvs/nessus镜像

# 拉取镜像
  docker pull leishianquan/awvs-nessus:v1
  # 启动
  docker run -it -d -p 13443:3443 -p 8834:8834 leishianquan/awvs-nessus:v1
  # 查看容器
  docker ps –a
  # 启动容器
  docker start container-id
  # 进入容器
  docker exec –it container-id /bin/bash

  # 进入容器后，启动nessus
  /etc/init.d/nessusd start

  # 访问扫描器地址和账号密码
  Nessus:
  https://127.0.0.1:8834/#/
  account:leishi/leishianquan

  Awvs13:
  https://127.0.0.1:13443/
  account:admin@admin.com/Admin123

  • 安装postgresql

sudo docker pull postgres:latest

docker volume create postgres-volume

docker run -d --name=postgres -p 5432:5432 -v postgres_docker_db:/var/lib/postgresql/data -e POSTGRES_PASSWORD=123456 postgres

  • 安装minio

docker run -dit --name minio -p 9000:9000 -p 9001:9001 -v ./minio/data/:/data -e MINIO_ROOT_USER="admin" -e MINIO_ROOT_PASSWORD="Pass@123123" bitnami/minio:2023.3.13

  • 安装dm8数据库

新版本 Docker 镜像中数据库默认用户名/密码为 SYSDBA/SYSDBA001

docker run -d -p 5236:5236 --restart=always --name dm8_01 --privileged=true -e PAGE_SIZE=16 -e LD_LIBRARY_PATH=/opt/dmdbms/bin -e INSTANCE_NAME=dm8_01 -v /data/dm8_01:/opt/dmdbms/data dm8_single:v8.1.2.128_ent_x86_64_ctm_pack4

  • 部署MySQL

部署5.7

sudo docker run --name mysql-server -e MYSQL_ROOT_PASSWORD=123456 -p 3306:3306  mysql:5.7

导入sql文件

sudo docker cp /tmp/apikiller.sql bteye-mysql:/tmp/apikiller.sql
sudo docker exec -it mysql-server mysql -uroot -p123456
source /tmp/apikiller.sql
